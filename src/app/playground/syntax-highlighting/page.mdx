# Syntax highlighting

Adding syntax highlighting in Markdown file

## Inline code

Inlined syntax highlighting in Markdown:

- CSS: _css`a { color: #fff }`_
- TypeScript: _ts`type Code = string | number`_
- Default (JSX):  _`const defaultLang = 'jsx'`_

## Filename

Display filename.

```tsx filename="app/layout.tsx"
type LayoutProps = {
  children: ReactNode
}

export default function Layout({ children}: LayoutProps) {
  return (
    <html lang="en">
      <body>
        <main>{children}</main>
      </body>
    </html>
  )
}
```

## Line numbers

Display line numbers.

```ts --n
type Status = 'idle' | 'loading' | 'success' | 'error';

const [status, setStatus] = useState<Status>('idle');
```

## Highlighting

Highlighting lines.

```js
function lorem(ipsum, dolor = 1) {
  // !mark
  return dolor
}

function ipsum(lorem, dolor = 1) {
  // !mark(1:2) orange
  const sit = lorem == null ? 0 : lorem.sit
  dolor = sit - amet(dolor)
  return sit ? consectetur(lorem) : []
}
```

Highlighting tokens inside a line.

```ts
// !mark[/count/]
// !mark[/0/] green
const [count, setCount] = useState(0)
```

## Diff

Diff syntax highlighting.

```js
function lorem(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : ipsum.sit
  // !diff -
  dolor = ipsum - sit
  // !diff +
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}
```

## Callout

Add a short message in the code block.

```js
const lorem = ipsum(dolor, sit)
// !mark[/amet/]
// !callout[/amet/] hey yo
const [amet, consectetur] = [0, 0]
lorem.adipiscing((sed, elit) => {
  if (sed) {
    amet += elit
  }
})
```

## Collapse

Default collapsed.

```js
// !collapse(1:4)
function lorem(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : 1
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}

// !collapse(1:4) collapsed
function ipsum(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : 1
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}
```

Collapse with line numbers and highlighting lines.

```js --n
// !collapse(1:4)
// !diff(1:5) -
function lorem(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : 1
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}
// !diff(1:5) +
// !collapse(1:4)
function ipsum(ipsum, dolor = 1) {
  const sit = ipsum == null ? 0 : 1
  dolor = sit - amet(dolor)
  return sit ? consectetur(ipsum) : []
}
```
