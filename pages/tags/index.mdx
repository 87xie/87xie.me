import Link from 'next/link'
import { useData } from 'nextra/hooks'

export function getStaticProps() {
  const NEXTRA_INTERNAL = Symbol.for('__nextra_internal__')
  const postsPage = globalThis[NEXTRA_INTERNAL]?.pageMap?.find(({ name }) => name === 'posts')
  const tags = postsPage?.children?.reduce((acc, post) => {
    if (typeof post?.frontMatter?.tag !== 'string') return acc
    const tagsFromFrontMatter = post.frontMatter.tag.split(',')
    tagsFromFrontMatter.forEach((tag) => {
      const _tag = tag.trim();
      if (!acc.includes(_tag)) acc.push(_tag)
    });
    return acc
  }, []) || []

  return Promise.resolve({
    props: {
      ssg: { tags }
    }
  })
}

export function Tags() {
  const data = useData()
  return (
    <div className="_flex _flex-wrap _gap-2">
      {data.tags.map((tag) => (
        <Link
          key={tag}
          href={`/tags/${tag}`}
          className="
            _inline_flex
            _px-1
            _select-none
            _rounded-md
            _bg-gray-200
            _text-md
            _text-gray-400
            _transition-colors
            hover:_bg-gray-300
            hover:_text-gray-500
            dark:_bg-gray-600
            dark:_text-gray-300
            dark:hover:_bg-gray-700
            dark:hover:_text-gray-200
          "
        >
          {tag}
        </Link>
      ))}
    </div>
  )
}

<Tags />
